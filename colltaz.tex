\section*{Language}
  \begin{align*}
    A &= \{ 0, 1 \} \\
    T &= A \ \cup \ \{ , \} \ \cup \  \{ \blm,  X \} \\
    L &= \{1, 10, 11, 100, 101, 110, 111, 1000, 1001, \ldots \} \\
    f(s) &= 1s,s \\
  \end{align*}
